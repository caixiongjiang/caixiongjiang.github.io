int bfs(int x, int y)
{
    node now, to;
    now.x = x, now.y = y, now.s = 0;
    queue <node>q;
    vis[x][y] = 1;
    q.push(now);
    while (!q.empty())
    {
        now = q.front();
        q.pop();
        if (now.x == e.x&&now.y == e.y)
            return now.s;
        for (int i = 0; i<4; i++)
        {
            int xx = now.x + go[i][0];
            int yy = now.y + go[i][1];
            if (xx >= 0 && xx<20 && yy >= 0 && yy<20 && maze[xx][yy] == 0 && !vis[xx][yy])
            {
                vis[xx][yy] = 1;
                to.x = xx, to.y = yy, to.s = now.s + 1;
                pre[xx * 20 + yy] = now.x * 20 + now.y;
                q.push(to);
            }
        }
    }
    return -1;
}
