	User     string `form:"user" json:"user" binding:"required"`
	Password string `form:"password" json:"password" binding:"required"`

func main() {
	router := gin.Default()

	// ÁªëÂÆöJSONÁöÑ‰æãÂ≠ê ({"user": "manu", "password": "123"})
	router.POST("/loginJSON", func(c *gin.Context) {
		var json Login

		if c.BindJSON(&json) == nil {
			if json.User == "manu" && json.Password == "123" {
				c.JSON(http.StatusOK, gin.H{"status": "you are logged in"})
			} else {
				c.JSON(http.StatusUnauthorized, gin.H{"status": "unauthorized"})
			}
		}
	})

	// ÁªëÂÆöÊôÆÈÄöË°®ÂçïÁöÑ‰æãÂ≠ê (user=manu&password=123)
	router.POST("/loginForm", func(c *gin.Context) {
		var form Login
		// Ê†πÊçÆËØ∑Ê±ÇÂ§¥‰∏≠ content-type Ëá™Âä®Êé®Êñ≠.
		if c.Bind(&form) == nil {
			if form.User == "manu" && form.Password == "123" {
				c.JSON(http.StatusOK, gin.H{"status": "you are logged in"})
			} else {
				c.JSON(http.StatusUnauthorized, gin.H{"status": "unauthorized"})
			}
		}
	})
	// ÁªëÂÆöÂ§öÂ™í‰ΩìË°®ÂçïÁöÑ‰æãÂ≠ê (user=manu&password=123)
	router.POST("/login", func(c *gin.Context) {

		// ‰Ω†ÂèØ‰ª•ÊòæÂºèÂ£∞ÊòéÊù•ÁªëÂÆöÂ§öÂ™í‰ΩìË°®ÂçïÔºö
		// c.BindWith(&form, binding.Form)
		// ÊàñËÄÖ‰ΩøÁî®Ëá™Âä®Êé®Êñ≠:
		if c.Bind(&form) == nil {
			if form.User == "user" && form.Password == "password" {
				c.JSON(200, gin.H{"status": "you are logged in"})
			} else {
				c.JSON(401, gin.H{"status": "unauthorized"})
			}
		}
	})
	// Listen and serve on 0.0.0.0:8080
	router.Run(":8080")
}
