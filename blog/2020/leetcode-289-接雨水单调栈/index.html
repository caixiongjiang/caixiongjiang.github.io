class Solution
{
public:
    int trap(vector<int> &height)
    {
        int ans = 0;
        stack<int> st;
        for (int i = 0; i < height.size(); i++)
        {
            while (!st.empty() && height[i] > height[st.top()])
            {
                int cur = height[st.top()];
                st.pop();
                int l = st.empty() ? -1 : st.top();
                ans += (l == -1) ? 0 : (min(height[l], height[i]) - cur) * (i - l - 1);
            }
            st.push(i);
        }
        return ans;
    }
};
