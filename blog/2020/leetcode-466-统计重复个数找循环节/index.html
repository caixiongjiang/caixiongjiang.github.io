class Solution
{
public:
    int getMaxRepetitions(string s1, int n1, string s2, int n2)
    {
        if (n1 == 0)
            return 0;
        int s1cnt = 0, s2cnt = 0, index = 0;

        // mp[index]={s1cnt,s2cnt}è¡¨ç¤ºï¼Œç¬¬indexä¸ªå­—ç¬¦åœ¨éå†s1cntæ¬¡s1å’Œs2cntæ¬¡s2å‡ºç°çš„
        unordered_map<int, pair<int, int>> mp;
        pair<int, int> pre, in;
        while (1)
        {
            s1cnt++;
            for (char ch : s1)
            {
                if (ch == s2[index])
                {
                    index++;
                    if (index == s2.size())
                    {
                        s2cnt++;
                        index = 0;
                    }
                }
            }
            // æ²¡æ‰¾åˆ°å¾ªç¯èŠ‚ï¼Œs1æ²¡äº†
            if (s1cnt == n1)
                return s2cnt / n2;

            if (mp.count(index))
            {
                auto [a, b] = mp[index];
                pre = {a, b};
                // æœ‰å¾ªç¯èŠ‚,ä»¥åçš„æ¯ (s1cnt-a)ä¸ªs1ä¸­åŒ…å«äº†(s2cnt-b)ä¸ªs2
                in = {s1cnt - a, s2cnt - b};
                break;
            }
            else
                mp[index] = {s1cnt, s2cnt};
        }
        int ans = pre.second + (n1 - pre.first) / in.first * in.second;
        // åŒ¹é…s1æœ«å°¾å‰©ä¸‹çš„
        int rest = (n1 - pre.first) % in.first;
        for (int i = 0; i < rest; ++i)
        {
            for (char ch : s1)
            {
                if (ch == s2[index])
                {
                    ++index;
                    if (index == s2.size())
                    {
                        ++ans;
                        index = 0;
                    }
                }
            }
        }
        return ans / n2;
    }
};
