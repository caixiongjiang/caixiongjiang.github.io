function[w] = mat2vector(mat, maxCnt)
    w = [];
    for i = 1 : maxCnt
        mat1 = mat(:, :, i);
        [x, y] = size(mat1);
        mat2 = mat1';
        mat3 = reshape(mat2, 1, x * y);
        w = [w; mat3];
    end
end
